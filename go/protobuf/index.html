1. 官方网站

https://developers.google.cn/protocol-buffers

2. Mac安装

brew install protobuf

3. 特点

  • 语言无关、平台无关：即 ProtoBuf 支持 Python、PHP、GO 等多种语言，支持多个平台
  • 高效：即比 XML 更小（3 ~ 10倍）、更快（20 ~ 100倍）、更为简单
  • 扩展性、兼容性好：可以更新数据结构，而不影响和破坏原有的旧程序

4. 安装protoc-gen-go

go install google.golang.org/protobuf/cmd/protoc-gen-go@latest

5. 定义消息类型

// 语法
syntax = "proto3";

// package，即包名声明符是可选的，用来防止不同的消息类型有命名冲突
package api;

// 指定报名
option go_package = "api/v1";

// 消息体
message User
{
    uint32 ID = 1;
    required string name = 2;
}

message UserList
{
    repeated User user = 1;
}

6.2. 复合类型

6.2.1. 枚举

message SearchRequest {
  string query = 1;
  int32 page_number = 2;
  int32 result_per_page = 3;
  enum Corpus {
    UNIVERSAL = 0;
    WEB = 1;
    IMAGES = 2;
    LOCAL = 3;
    NEWS = 4;
    PRODUCTS = 5;
    VIDEO = 6;
  }
  Corpus corpus = 4;
}
  • 枚举的第一个常量标识符必须是0，也是默认值
  • 别名，默认情况下，常量标识符不允许重复，如果要重复，需要将allow_alias选项设置为true，这样就可以定义别名。
    message MyMessage1 {
enum EnumAllowingAlias {
  option allow_alias = true;
  UNKNOWN = 0;
  STARTED = 1;
  RUNNING = 1;
}
}
message MyMessage2 {
enum EnumNotAllowingAlias {
  UNKNOWN = 0;
  STARTED = 1;
  // RUNNING = 1;  // Uncommenting this line will cause a compile error inside Google and a warning message outside.
}
}

    6.2.2. 嵌套类型

    可以在消息类型中定义和使用其它消息类型
    message Result {
  string url = 1;
  string title = 2;
  repeated string snippets = 3;
}
message SearchResponse {
repeated Result results = 1;
}

    message SearchResponse {
message Result {
  string url = 1;
  string title = 2;
  repeated string snippets = 3;
}
repeated Result results = 1;
}
    如果您想在其父消息类型之外重用此消息类型，可以使用格式（Parent.Type）：
    message SomeOtherMessage {
SearchResponse.Result result = 1;
}

    6.2.3. 任意类型（Any）

    需要导入google/protobuf/any.proto ```go import "google/protobuf/any.proto";

message ErrorStatus { string message = 1; repeated google.protobuf.Any details = 2; }

### Oneof
```go
message SampleMessage {
  oneof test_oneof {
    string name = 4;
    SubMessage sub_message = 9;
  }
}
  • 可以添加任何类型的字段，但map字段和repeated字段除外；
  • 设置 oneof 字段将自动清除 oneof 的所有其他成员。因此，如果您设置了多个 oneof 字段，则只有您设置的最后一个字段仍有值

  • 6.2.4. Map

    map类型的语法格式如下：
    map<key_type, value_type> map_field = N;
  • key_type可以是任何整数或字符串类型;
  • value_type可以是任何类型，除了另一个map;
  • 地图字段不能是repeated

7. 字段编码

消息定义中的每个字段都有一个唯一的编号。这些字段编号用于在消息二进制格式中标识您的字段，并且在使用您的消息类型后不应更改。

请注意:

1 到 15 范围内的字段编号需要一个字节进行编码，包括字段编号和字段类型。 16 到 2047 范围内的字段编号占用两个字节。 因此，您应该为非常频繁出现的消息元素保留数字 1 到 15。请记住为将来可能添加的频繁出现的元素留出一些空间。

最小字段编号是 1，最大的是 2 29 - 1，即 536,870,911。您也不能使用数字 19000 到 19999（FieldDescriptor::kFirstReservedNumber到FieldDescriptor::kLastReservedNumber），因为它们是为 Protocol Buffers 实现保留的. 同样，您不能使用任何以前保留的字段编号

8. 字段规则

  • singular：格式良好的消息可以有零个或一个此字段（但不能超过一个）。这是 proto3 语法的默认字段规则。
  • repeated：该字段可以在格式良好的消息中重复任意次数（包括零次）。重复值的顺序将被保留。

10.1. 编译

使用命令：protoc --go_out=. user.proto，其它参数如下：

参数 说明
--proto_path或者-I 指定 import 路径，可以指定多个参数，编译时按顺序查找，不指定时默认查找当前目录
--go_out 指定pb文件输出路径
--gogo_out 指定pb文件输出路径，这个要使用gogoproto包，可以添加参数验证
--go_opt 指定参数，比如--go_opt=paths=source_relative就是表明生成文件输出使用相对路径
--go-http_out 指定http文件输出路径，这个要使用kratos框架
--go-grpc_out 指定grpc文件输出路径，这个要使用grpc包

user.proto代码如下：

syntax = "proto3";

import "github/gogo/protobuf/gogoproto/gogo.proto";

package api;

option go_package = "api/v1";

message UserListRequest {
  string name = 1 [(gogoproto.jsontag) = 'username', (gogoproto.moretags) = 'form:"username" validate:"required"'];
  int32 age = 2;
}

message User {
  string name = 1;
  int32 age = 2;
  int32 sex = 3;
  string city = 4;
}

message UserListResponse {
  repeated User list = 1;
}


service SearchService {
  rpc Search(UserListRequest) returns (UserListResponse);
}

user_service.go代码如下：

package main

import (
    "fmt"
    v1 "proto/api/v1"

    "github.com/gogo/protobuf/proto"
    "io/ioutil"
    "net/http"
)

func main() {
    //单独写回调函数
    http.HandleFunc("/post", getHandler)
    // addr：监听的地址
    // handler：回调函数
    _ = http.ListenAndServe("127.0.0.1:9090", nil)
}

func getHandler(w http.ResponseWriter, r *http.Request) {
    b, err := ioutil.ReadAll(r.Body)
    fmt.Println(r.PostForm.Get("name"))

    if err != nil {
        fmt.Println("读取内容错误:", err.Error())
    }
    req := &v1.UserListRequest{}
    err = proto.Unmarshal(b, req)

    if err != nil {
        fmt.Println("内容解析错误:", err.Error())
    }
    fmt.Println(req)

    resp := &v1.UserListResponse{
        List: []*v1.User{
            {
                Name: "张三",
                Age:  18,
                Sex:  1,
                City: "北京",
            },
            {
                Name: "李四",
                Age:  20,
                Sex:  0,
                City: "西安",
            },
            {
                Name: "王五",
                Age:  28,
                Sex:  1,
                City: "河南",
            },
        },
    }

    protoResp, _ := proto.Marshal(resp)
    w.Write(protoResp)
}

user_client.go代码如下：

package main

import (
    "bytes"
    "fmt"
    "github.com/gogo/protobuf/proto"
    "io/ioutil"
    "net/http"
    v1 "proto/api/v1"
)

func main() {
    params := &v1.UserListRequest{}
    params.Name = "lsc"
    params.Age = 28

    //Post请求示例
    url := "http://127.0.0.1:9090/post"
    contentType := "application/protobuf"
    protoBody, err := proto.Marshal(params)
    if err != nil {
        panic("数据解析失败:" + err.Error())
    }

    resp, err := http.Post(url, contentType, bytes.NewReader(protoBody))
    if err != nil {
        fmt.Println("post failed, err:", err)
        return
    }
    defer resp.Body.Close()
    b, err := ioutil.ReadAll(resp.Body)

    if err != nil {
        fmt.Println("get resp failed, err", err)
        return
    }
    fmt.Println(string(b)) //Post请求示例

}
